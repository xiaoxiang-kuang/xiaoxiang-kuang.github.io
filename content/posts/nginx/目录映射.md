```nginx
server {
    listen 20000;
    location / {
        root /opt/share;
        //开启自动映射目录
        autoindex on;
        charset utf-8;
        //时间设置为本地时间
        autoindex_localtime on;
        //通过字节精确表示大小
        autoindex_exact_size on;
    }
}
```
